% Directed graphical model (plate diagram) for the CSPF2 model
% Covariate Seeded Poisson Factorization with grouped design-adaptive shrinkage
%
% Compile standalone:  pdflatex CSPF2_graphical_model.tex
% Or include in a paper via \input{...} (remove \documentclass and \begin{document})

\documentclass[tikz, border=15pt]{standalone}
\usepackage{amsmath, amssymb, bm}
\usetikzlibrary{arrows.meta, backgrounds}

\begin{document}
\begin{tikzpicture}[
  % --- Node styles ---
  latent/.style = {
    circle, draw, thick, minimum size=12mm,
    inner sep=0pt, font=\small
  },
  obs/.style = {
    circle, draw, thick, fill=gray!30, minimum size=12mm,
    inner sep=0pt, font=\small
  },
  det/.style = {
    circle, draw, thick, double, double distance=1.5pt,
    minimum size=12mm, inner sep=0pt, font=\small
  },
  param/.style = {
    fill=black, circle, inner sep=0pt, minimum size=4pt
  },
  % --- Arrow style ---
  arr/.style = {-{Stealth[length=5pt]}, thick},
]

% ================================================================
%  LAYER 1 — SHRINKAGE HYPERPARAMETERS  (top)
% ================================================================

% -- Global shrinkage (topic-level: τ²_k) --
\node[param, label={[font=\scriptsize]160:$a_{\rho\tau}$}]
  (arhotau)  at (-0.65, 10.3) {};
\node[param, label={[font=\scriptsize]20:$b_{\rho\tau}$}]
  (brhotau)  at ( 0.65, 10.3) {};
\node[latent] (rhotau)  at (0, 9.2)    {$\rho_k^{(\tau)}$};

\node[param, label={[font=\scriptsize]180:$a_{\tau}$}]
  (atau)     at (-1.4, 7.7) {};
\node[latent] (tau)     at (0, 7.7)    {$\tau_k^2$};

% -- Local shrinkage (group × topic: δ²_{gk}) --
\node[param, label={[font=\scriptsize]160:$a_{\rho\delta}$}]
  (arhodelta) at (3.85, 10.3) {};
\node[param, label={[font=\scriptsize]20:$b_{\rho\delta}$}]
  (brhodelta) at (5.15, 10.3) {};
\node[latent] (rhodelta) at (4.5, 9.2)  {$\rho_{gk}^{(\delta)}$};

\node[param, label={[font=\scriptsize]0:$a_{\delta}$}]
  (adelta)   at (6.1, 7.7) {};
\node[latent] (delta)   at (4.5, 7.7)  {$\delta_{gk}^2$};


% ================================================================
%  LAYER 2 — REGRESSION COEFFICIENTS
% ================================================================

% -- Intercept --
\node[param, label={[font=\scriptsize]90:$s^2_{\lambda_0}$}]
  (slambda)  at (-3, 6.3) {};
\node[latent] (lambda0) at (-3, 5.2)   {$\lambda_{0k}$};

% -- Group coefficients --
\node[latent] (lambdagk) at (2.5, 6.0) {$\lambda_{gk}$};
\node[obs]    (Xg)       at (5.8, 6.0) {$\bm{X}_g$};


% ================================================================
%  LAYER 3 — DOCUMENT–TOPIC LINK
% ================================================================

\node[obs] (xd) at (-3, 3.2)    {$\bm{x}_d$};
\node[det] (mu) at ( 0.3, 3.2)  {$\mu_{\theta,kd}$};

\node[param, label={[font=\scriptsize]180:$b_{\theta}$}]
  (btheta) at (-1.3, 1.6) {};
\node[latent] (theta) at (0.3, 1.6)  {$\theta_{dk}$};


% ================================================================
%  LAYER 4 — TOPIC–WORD (seed decomposition)
% ================================================================

\node[param, label={[font=\scriptsize]90:$a_{\beta^\star}\!,b_{\beta^\star}$}]
  (abbstar)  at (7, 4.5) {};
\node[latent] (bstar)   at (7, 3.2)    {$\beta_{kv}^\star$};

\node[param, label={[font=\scriptsize]90:$a_{\widetilde{\beta}}\!,b_{\widetilde{\beta}}$}]
  (abbtilde) at (9.5, 4.5) {};
\node[latent] (btilde)  at (9.5, 3.2)  {$\widetilde{\beta}_{kv}$};

\node[det] (beta) at (8.25, 1.6)  {$\beta_{kv}$};


% ================================================================
%  LAYER 5 — OBSERVATION
% ================================================================

\node[obs] (y) at (4, 0) {$y_{dv}$};


% ================================================================
%  EDGES
% ================================================================

% Global shrinkage chain
\draw[arr] (arhotau)  -- (rhotau);
\draw[arr] (brhotau)  -- (rhotau);
\draw[arr] (rhotau)   -- (tau);
\draw[arr] (atau)     -- (tau);

% Local shrinkage chain
\draw[arr] (arhodelta) -- (rhodelta);
\draw[arr] (brhodelta) -- (rhodelta);
\draw[arr] (rhodelta)  -- (delta);
\draw[arr] (adelta)    -- (delta);

% Shrinkage → group coefficients
\draw[arr] (tau)   -- (lambdagk);
\draw[arr] (delta) -- (lambdagk);
\draw[arr] (Xg)   -- (lambdagk);

% Intercept prior
\draw[arr] (slambda) -- (lambda0);

% Coefficients → deterministic link
\draw[arr] (lambda0)  -- (mu);
\draw[arr] (lambdagk) -- (mu);
\draw[arr] (xd)       -- (mu);

% Link → document-topic intensity
\draw[arr] (mu)     -- (theta);
\draw[arr] (btheta) -- (theta);

% Document-topic → observation
\draw[arr] (theta) -- (y);

% Topic-word priors
\draw[arr] (abbstar)  -- (bstar);
\draw[arr] (abbtilde) -- (btilde);

% Seed decomposition → deterministic sum
\draw[arr] (bstar)  -- (beta);
\draw[arr] (btilde) -- (beta);

% Topic-word → observation
\draw[arr] (beta) -- (y);


% ================================================================
%  PLATES  (explicit coordinates matching the Python renderer)
% ================================================================
% Using explicit \draw rectangles instead of fit to ensure identical
% plate boundaries across all LaTeX engines (pdflatex, Overleaf, etc.)

\begin{scope}[on background layer]

  % --- Topic plate  k = 1, ..., K ---
  % Contains: τ, ρ^(τ), λ_{0k}, g-plate, μ, θ, β*, β̃, β — but NOT y_{dv}
  \draw[thick, dashed, rounded corners=5pt, draw=orange!70!black, fill=orange!5]
    (-4.8, 0.8) rectangle ++(16.0, 10.4);
  \node[font=\small, text=orange!70!black, anchor=north west]
    at (-4.7, 11.1) {$k=1,\ldots,K$};

  % --- Group plate  g = 1, ..., G ---
  % Contains: ρ^(δ)_{gk}, δ²_{gk}, λ_{gk}, X_g
  \draw[thick, dashed, rounded corners=5pt, draw=blue!70!black, fill=blue!5]
    (1.7, 5.2) rectangle ++(5.8, 5.8);
  \node[font=\small, text=blue!70!black, anchor=south east]
    at (7.4, 5.3) {$g=1,\ldots,G$};

  % --- Document plate  d = 1, ..., D ---
  % Contains: x_d, μ, θ, y_{dv}
  \draw[thick, dashed, rounded corners=5pt, draw=green!50!black, fill=green!6]
    (-3.8, -0.7) rectangle ++(9.3, 4.8);
  \node[font=\small, text=green!50!black, anchor=south west]
    at (-3.7, -0.6) {$d=1,\ldots,D$};

  % --- Vocabulary plate  v = 1, ..., V ---
  % Contains: β*, β̃, β, y_{dv}
  \draw[thick, dashed, rounded corners=5pt, draw=red!70!black, fill=red!5]
    (3.2, -0.7) rectangle ++(7.3, 6.0);
  \node[font=\small, text=red!70!black, anchor=south east]
    at (10.4, -0.6) {$v=1,\ldots,V$};

\end{scope}


% ================================================================
%  LEGEND
% ================================================================
\matrix[draw, rounded corners, fill=white,
        column sep=8pt, row sep=3pt,
        inner sep=6pt, font=\scriptsize,
        anchor=north west] at (-5.5, 10.8) {
  \node[obs,    minimum size=6mm] {}; & \node[anchor=west]{Observed};       \\
  \node[latent, minimum size=6mm] {}; & \node[anchor=west]{Latent};         \\
  \node[det,    minimum size=6mm] {}; & \node[anchor=west]{Deterministic};  \\
  \node[param]                    {}; & \node[anchor=west]{Fixed parameter};\\
};

\end{tikzpicture}
\end{document}
